name: üöÄ Advanced ESP32 CI

on:
  push:
    branches: [main, develop, release/*]
  pull_request:
    branches: [main, develop]
  workflow_dispatch:
    inputs:
      clean_build:
        description: 'Clean build (no cache)'
        required: false
        default: false
        type: boolean
      run_security:
        description: 'Run security checks'
        required: false
        default: true
        type: boolean

env:
  PROJECT_DIR: examples/esp32
  TOOLS_DIR: hf-espidf-project-tools

jobs:
  # üîç Matrix generation and validation
  validate:
    runs-on: ubuntu-latest
    outputs:
      matrix: $
    steps:
      - uses: actions/checkout@v4
        with:
          submodules: recursive
      
      - name: Validate configuration
        run: |
          cd $
          python3 $/generate_matrix.py --validate --verbose

      - name: Generate build matrix
        id: matrix
        run: |
          cd $
          MATRIX=$(python3 $/generate_matrix.py)
          echo "matrix=${MATRIX}" >> "$GITHUB_OUTPUT"
          echo "Generated matrix with $(echo "$MATRIX" | jq '.include | length') build combinations"

  # üèóÔ∏è Build firmware (parallel matrix)
  build:
    needs: validate
    uses: N3b3x/hf-espidf-ci-tools/.github/workflows/build.yml@v1
    with:
      project_dir: $
      project_tools_dir: $
      clean_build: $

  # üõ°Ô∏è Security audit
  security:
    if: $
    uses: N3b3x/hf-espidf-ci-tools/.github/workflows/security.yml@v1
    with:
      project_dir: $
      project_tools_dir: $
      scan_type: "all"
      run_codeql: true
      codeql_languages: "cpp,python"

  # üìä Build summary and notifications
  summary:
    needs: [build, security]
    if: always()
    runs-on: ubuntu-latest
    steps:
      - name: Build Summary
        run: |
          echo "## üöÄ Build Summary" >> $GITHUB_STEP_SUMMARY
          echo "" >> $GITHUB_STEP_SUMMARY
          echo "| Job | Status |" >> $GITHUB_STEP_SUMMARY
          echo "|-----|--------|" >> $GITHUB_STEP_SUMMARY
          echo "| üèóÔ∏è Build | $ |" >> $GITHUB_STEP_SUMMARY
          echo "| üõ°Ô∏è Security | $ |" >> $GITHUB_STEP_SUMMARY

name: üì¶ Release CI

on:
  push:
    branches: [release/*, main]
    tags: ['v*']
  workflow_dispatch:
    inputs:
      release_version:
        description: 'Release version (e.g., v1.0.0)'
        required: true
        type: string

env:
  PROJECT_DIR: examples/esp32
  RELEASE_VERSION: $

jobs:
  # Comprehensive build for release
  build-release:
    uses: N3b3x/hf-espidf-ci-tools/.github/workflows/build.yml@v1
    with:
      project_dir: $
      clean_build: true  # Clean build for release

  # Full security audit
  security-release:
    uses: N3b3x/hf-espidf-ci-tools/.github/workflows/security.yml@v1
    with:
      project_dir: $
      scan_type: "all"
      run_codeql: true
      codeql_languages: "cpp,python"

  # Create release artifacts
  create-release:
    needs: [build-release, security-release]
    if: startsWith(github.ref, 'refs/tags/')
    runs-on: ubuntu-latest
    steps:
      - name: Download build artifacts
        uses: actions/download-artifact@v4
        with:
          pattern: fw-*
          merge-multiple: true
          path: artifacts/

      - name: Create Release
        uses: actions/create-release@v1
        env:
          GITHUB_TOKEN: $
        with:
          tag_name: $
          release_name: Release $
          body: |
            ## üöÄ HardFOC ESP32 Release $
            
            ### üì¶ Firmware Binaries
            - All firmware binaries are attached to this release
            - Built with ESP-IDF v5.5
            - Target: ESP32-C6
            
            ### üìö Documentation
            - [Documentation](https://$.github.io/$/)
            
            ### üõ°Ô∏è Security
            - Full security audit completed
            - CodeQL analysis passed
            - No known vulnerabilities
          draft: false
          prerelease: false

      - name: Upload Release Assets
        uses: actions/upload-release-asset@v1
        env:
          GITHUB_TOKEN: $
        with:
          upload_url: $
          asset_path: artifacts/
          asset_name: firmware-$.zip
          asset_content_type: application/zip

name: üõ°Ô∏è Security-First CI

on:
  push:
    branches: [main, security/*]
  pull_request:
    branches: [main]
  schedule:
    - cron: '0 2 * * 1'  # Weekly security scan

jobs:
  # Security audit (runs first)
  security-audit:
    uses: N3b3x/hf-espidf-ci-tools/.github/workflows/security.yml@v1
    with:
      project_dir: examples/esp32
      scan_type: "all"
      run_codeql: true
      codeql_languages: "cpp,python"

  # Build only if security passes
  build-secure:
    needs: security-audit
    uses: N3b3x/hf-espidf-ci-tools/.github/workflows/build.yml@v1
    with:
      project_dir: examples/esp32
      clean_build: true

  # Additional security checks
  security-scan:
    needs: security-audit
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Run Trivy vulnerability scanner
        uses: aquasecurity/trivy-action@master
        with:
          scan-type: 'fs'
          scan-ref: '.'
          format: 'sarif'
          output: 'trivy-results.sarif'

      - name: Upload Trivy scan results
        uses: github/codeql-action/upload-sarif@v2
        with:
          sarif_file: 'trivy-results.sarif'
